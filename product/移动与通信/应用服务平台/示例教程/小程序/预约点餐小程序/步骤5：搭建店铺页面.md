        this.setData({

        })
    },
    bindPickerChange1: function(e) {

          index1: e.detail.value
        })

      console.log(e)
      this.search(e.detail.value)

    //点击完成按钮时触发

      this.search(e.detail.value)
    }, 
    search(key) {
      var that = this;
      //从本地缓存中异步获取指定 key 的内容
      var list = wx.getStorage({
        key: 'list',
        //从Storage中取出存储的数据
        success: function (res) {
          // console.log(res)
          if (key == '') { //用户没有输入时全部显示
            that.setData({
              list: res.data
            })
            return;
          }
          var arr = []; //临时数组，用于存放匹配到的数组
          for (let i in res.data) {
            res.data[i].show = false; //所有数据隐藏
            if (res.data[i].name.indexOf(key) >= 0) {
              res.data[i].show = true; //让匹配到的数据显示
              arr.push(res.data[i])
            }
          }
          if (arr.length == 0) {
            that.setData({
              list: [{
                show: true,
                name: '没有相关数据！'
              }]
            })
          } else {
            that.setData({
              list: arr
            })
          }
        },
      })
    },
    bindPickerChange2: function(e) {
        console.log('picker发送选择改变，携带值为', e.detail.value)
        this.setData({
          index1: e.detail.value
        })
    },
    /**
     * 生命周期函数--监听页面加载

             })

这里我们考虑到，我们点击相应的商家，应该显示出来商家的商品，这里我们在查询数据库时候要构建查询指令，这里我们在用户在商家之后将商家id传到页面，我们在onloadl里面获取到保存在list_id里面。然后我们使用`where` 方法我们可以构造复杂的查询条件完成复杂的查询任务。

> | 查询指令 | 说明                 |
> | :------- | :------------------- |
> | eq       | 等于                 |
> | neq      | 不等于               |
> | lt       | 小于                 |
> | lte      | 小于或等于           |
> | gt       | 大于                 |
> | gte      | 大于或等于           |
> | in       | 字段值在给定数组中   |
> | nin      | 字段值不在给定数组中 |
